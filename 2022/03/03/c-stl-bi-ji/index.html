使用auto必须注意的事:即auto永远不会给你生成引用，你必须自己加才行

#include<bits/stdc++.h>
using namespace std; 
class Container
{
public:
    std::vector<int>& GetData()   {
        return data_;
    }
    std::vector<int> data_;
};


int main()
{
    Container c;
    c.data_ = { 0,1,2,3 };
    auto data = c.GetData();//这里data实际上是一个std::vector<int>而非 std::vector<int>&,
	//引用使得data和c.data_共地址,不管程序在任何时候，只要c.data_变，data就得变 
    c.data_.push_back(10);
    for (auto& it : data)  //若不加&不会输出10 
        std::cout << it << std::endl;
        for (auto it2 : c.data_)  
        std::cout << it2 << std::endl;//
    return 0;
}
